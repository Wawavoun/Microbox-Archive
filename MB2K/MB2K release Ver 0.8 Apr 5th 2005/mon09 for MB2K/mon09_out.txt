   decode_and_data_mux : process (addr(12 downto 11),
                                  cs0, cs1, cs2 )  
   begin
   	we <= not rw;
   	case addr(12 downto 11) is
	
		-- $F800 - $FFFF
		when "11" => 
		   rdata <= rdata2;
		   cs2 <= cs; 
		   cs1 <= '0';
		   cs0 <= '0';
 
		-- $F000 - $F7FF
		when "10" => 
		   rdata <= rdata1;
		   cs2 <= '0'; 
		   cs1 <= cs;
		   cs0 <= '0';

		-- $E800 - EFFF
		when "01" => 
		   rdata <= rdata0; 
		   cs2 <= '0'; 
		   cs1 <= '0';
		   cs0 <= cs;

		when others =>
